## Overview

Muun's Emergency Kit is a PDF document containing encrypted wallet keys and output descriptors for fund recovery. It enables users to recover their funds even if Muun servers are unavailable.

## Emergency Kit Structure

### Input Parameters

```go
type Input struct {
    FirstEncryptedKey  string  // User's encrypted key
    FirstFingerprint   string  // User key fingerprint
    SecondEncryptedKey string  // Muun's encrypted key
    SecondFingerprint  string  // Muun key fingerprint
    Version            int     // Kit version
}
```

### Output Structure

```go
type Output struct {
    HTML             string  // Rendered HTML
    VerificationCode string  // 6-digit code
}
```

```go Go (libwallet/emergencykit/emergencykit.go:43)
func GenerateHTML(
    params *Input,
    lang string,
) (*Output, error) {
    // Generate deterministic verification code
    verificationCode := generateDeterministicCode(params)
    
    // Render output descriptors if fingerprints provided
    var descriptors string
    if params.hasFingerprints() {
        descriptors = GetDescriptorsHTML(&DescriptorsData{
            FirstFingerprint:  params.FirstFingerprint,
            SecondFingerprint: params.SecondFingerprint,
        })
    }
    
    // Render page content
    content, err := render("EmergencyKitContent", lang, &contentData{
        FirstEncryptedKey:  params.FirstEncryptedKey,
        SecondEncryptedKey: params.SecondEncryptedKey,
        VerificationCode:   verificationCode,
        CurrentDate:        formatDate(time.Now(), lang),
        Descriptors:        descriptors,
        IconHelp:           iconHelp,
        IconPadlock:        iconPadlock,
    })
    if err != nil {
        return nil, fmt.Errorf("failed to render content: %w", err)
    }
    
    // Render complete HTML page
    page, err := render("EmergencyKitPage", lang, &pageData{
        Css:     css,
        Content: content,
    })
    if err != nil {
        return nil, fmt.Errorf("failed to render page: %w", err)
    }
    
    return &Output{
        HTML:             page,
        VerificationCode: verificationCode,
    }, nil
}
```

## Verification Code

### Deterministic Generation

The verification code is derived from the encrypted keys:

```go
func generateDeterministicCode(params *Input) string {
    // Use only second key (Muun key) for stability
    // Note: This compensates for client bugs with first key serialization
    inputMaterial := params.SecondEncryptedKey + strconv.Itoa(params.Version)
    
    // Compute SHA-256 hash
    inputHash := sha256.Sum256([]byte(inputMaterial))
    
    // Extract 6-digit code from hash
    var code string
    for _, b := range inputHash[:6] {
        code += strconv.Itoa(int(b) % 10)
    }
    
    return code
}
```

### iOS Implementation

```swift
import CryptoKit

func generateVerificationCode(
    encryptedKey: String,
    version: Int
) -> String {
    let material = encryptedKey + String(version)
    let hash = SHA256.hash(data: material.data(using: .utf8)!)
    
    var code = ""
    for byte in hash.prefix(6) {
        code += String(Int(byte) % 10)
    }
    
    return code
}
```

## Key Encryption

### Encrypt HD Key

Keys are encrypted with the recovery code:

```swift
func encryptKey(
    _ publicKey: LibwalletHDPublicKey,
    with recoveryCode: String
) throws -> String {
    // Serialize extended public key
    let xpub = publicKey.string()
    let xpubData = xpub.data(using: .utf8)!
    
    // Derive encryption key from recovery code
    let challengeKey = try LibwalletRecoveryCodeToKey(
        recoveryCode,
        "muun"
    )
    
    // Encrypt with challenge key
    let encrypter = challengeKey.encrypter()
    let encrypted = try encrypter.encrypt(xpubData)
    
    return encrypted
}
```

```go
func formatDate(t time.Time, lang string) string {
    if lang == "en" {
        return t.Format("January 2, 2006")
    } else {
        year, month, day := t.Date()
        monthName := spanishMonthNames[month-1]
```

## PDF Generation

### Convert HTML to PDF

```swift
import WebKit

class EmergencyKitPDFGenerator {
    func generatePDF(html: String) async throws -> Data {
        let webView = WKWebView()
        
        // Load HTML
        webView.loadHTMLString(html, baseURL: nil)
        
        // Wait for load
        try await Task.sleep(nanoseconds: 1_000_000_000)
        
        // Create PDF
        let pdfConfiguration = WKPDFConfiguration()
        pdfConfiguration.rect = CGRect(
            x: 0, y: 0,
            width: 8.5 * 72,  // Letter size
            height: 11 * 72
        )
        
        return try await webView.pdf(configuration: pdfConfiguration)
    }
}

// Usage
let generator = EmergencyKitPDFGenerator()
let pdfData = try await generator.generatePDF(html: output.html)

// Save to file
let fileURL = FileManager.default.temporaryDirectory
    .appendingPathComponent("emergency-kit.pdf")
try pdfData.write(to: fileURL)
```

## Complete Generation Flow

### iOS Implementation

```swift
class EmergencyKitManager {
    func generateEmergencyKit(
        recoveryCode: String,
        userKey: LibwalletHDPublicKey,
        muunKey: LibwalletHDPublicKey,
        language: String = "en"
    ) async throws -> (pdf: Data, verificationCode: String) {
        // 1. Encrypt keys with recovery code
        let userEncrypted = try encryptKey(userKey, with: recoveryCode)
        let muunEncrypted = try encryptKey(muunKey, with: recoveryCode)
        
        // 2. Get key fingerprints
        let userFingerprint = userKey.fingerprintHex()
        let muunFingerprint = muunKey.fingerprintHex()
        
        // 3. Create input
        let input = LibwalletEmergencyKitInput(
            firstEncryptedKey: userEncrypted,
            firstFingerprint: userFingerprint,
            secondEncryptedKey: muunEncrypted,
            secondFingerprint: muunFingerprint,
            version: 2
        )
        
        // 4. Generate HTML
        let output = try LibwalletGenerateEmergencyKitHTML(input, language)
        
        // 5. Convert to PDF
        let generator = EmergencyKitPDFGenerator()
        let pdfData = try await generator.generatePDF(html: output.html)
        
        return (pdfData, output.verificationCode)
    }
    
    func saveEmergencyKit(
        _ pdfData: Data,
        verificationCode: String
    ) throws -> URL {
        let filename = "muun-emergency-kit-\(verificationCode).pdf"
        let documentsURL = FileManager.default.urls(
            for: .documentDirectory,
            in: .userDomainMask
        )[0]
        let fileURL = documentsURL.appendingPathComponent(filename)
        
        try pdfData.write(to: fileURL)
        
        return fileURL
    }
    
    func shareEmergencyKit(_ fileURL: URL) -> UIActivityViewController {
        let activityVC = UIActivityViewController(
            activityItems: [fileURL],
            applicationActivities: nil
        )
        
        // Exclude some activities
        activityVC.excludedActivityTypes = [
            .addToReadingList,
            .assignToContact,
            .postToFacebook,
            .postToTwitter
        ]
        
        return activityVC
    }
}
```

    Never log or display unencrypted keys:
    ```swift
    // ❌ WRONG - exposes private keys
    print("Private key: \(privateKey.string())")
    
    // ✅ CORRECT - only use encrypted

    print("Encrypted key: \(encrypted.prefix(20))...")
    ```

    ```swift
        // Attempt decryption
    ```

## Error Handling

```swift
enum EmergencyKitError: Error {
    case missingRecoveryCode
    case missingKeys
    case encryptionFailed
    case htmlGenerationFailed
    case pdfGenerationFailed
    case invalidVersion
    
    var localizedDescription: String {
        switch self {
        case .missingRecoveryCode:
            return "Recovery code is required"
        case .missingKeys:
            return "Both user and Muun keys are required"
        case .encryptionFailed:
            return "Failed to encrypt keys"
        case .htmlGenerationFailed:
            return "Failed to generate HTML"
        case .pdfGenerationFailed:
            return "Failed to generate PDF"
        case .invalidVersion:
            return "Unsupported emergency kit version"
        }
    }
}
```
