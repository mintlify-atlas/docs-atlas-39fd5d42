## Overview

Libwallet provides comprehensive Lightning invoice parsing using the BOLT 11 specification. It supports unified QR codes with both on-chain and Lightning payment options.

## Invoice Structure

```go
type Invoice struct {
    RawInvoice      string

    Network         *Network
    MilliSat        string
    Destination     []byte
    PaymentHash     []byte

    Description     string
    Sats            int64
}
```

## Parsing Invoices

### Parse Lightning Invoice

<CodeGroup>

```go Go (libwallet/invoice.go:26)
func ParseInvoice(
    rawInput string,
    network *Network,
) (*Invoice, error) {
    _, components := buildUriFromString(rawInput, lightningScheme)
    if components == nil {
        return nil, errors.Errorf(ErrInvalidInvoice, "failed to parse uri %v", rawInput)
    }
    
    if components.Scheme != "lightning" {
        return nil, errors.Errorf(ErrInvalidInvoice, "invalid scheme %v", components.Scheme)
    }
    
    invoice := components.Opaque
    
    // When URIs are lightning:// the invoice comes in host
    if len(invoice) == 0 {
        invoice = components.Host
    }
    
    parsedInvoice, err := zpay32.Decode(invoice, network.network)
    if err != nil {
        return nil, errors.Errorf(ErrInvalidInvoice, "Couldn't parse invoice: %w", err)
    }
    
    var fallbackAdd *MuunPaymentURI
    if parsedInvoice.FallbackAddr != nil {
        fallbackAdd, err = GetPaymentURI(
            parsedInvoice.FallbackAddr.String(),
            network,
        )
        if err != nil {
            return nil, errors.Errorf(ErrInvalidInvoice, "Couldn't get address: %w", err)
        }
    }
    
    var description string
    if parsedInvoice.Description != nil {
        description = *parsedInvoice.Description
    }
    
    var milliSats string
    var sats int64
    if parsedInvoice.MilliSat != nil {
        milliSat := uint64(*parsedInvoice.MilliSat)
        milliSats = fmt.Sprintf("%v", milliSat)
        sats = int64(milliSat / 1000)
    }
    
    return &Invoice{
        RawInvoice:      invoice,
        FallbackAddress: fallbackAdd,
        Network:         network,
        MilliSat:        milliSats,
        Destination:     parsedInvoice.Destination.SerializeCompressed(),
        PaymentHash:     parsedInvoice.PaymentHash[:],
        Expiry:          parsedInvoice.Timestamp.Unix() + int64(parsedInvoice.Expiry().Seconds()),
        Description:     description,
        Sats:            sats,
    }, nil
}
```

## Amount Validation

### Unified QR Amount Matching

When both on-chain and Lightning payment options exist, amounts must match:

```go
if invoice != nil {
    if invoice.Sats != 0 {
        invoiceAmount := decimal.NewFromInt(invoice.Sats).
            Div(decimal.NewFromInt(100_000_000)).String()
        
        // Only mark as invalid if both amounts exist and differ
        if amount != "" {
            if invoiceAmount != amount {
                return nil, errors.New(ErrInvalidURI, "Amount mismatch")
            }
        }
    }
}
```

### iOS Amount Validation

```swift
func validateUnifiedQR(
    _ paymentURI: LibwalletMuunPaymentURI
) throws {
    guard let invoice = paymentURI.invoice else {
        return // No invoice, nothing to validate
    }
    
    // If invoice has amount, it must match on-chain amount
    if invoice.sats > 0,
       let onChainAmount = paymentURI.amount,
       !onChainAmount.isEmpty {
        let invoiceAmountBTC = Decimal(invoice.sats) / 100_000_000
        let onChainAmountBTC = Decimal(string: onChainAmount)
        
        guard invoiceAmountBTC == onChainAmountBTC else {
            throw PaymentError.amountMismatch
        }
    }
}
```

## Invoice Expiry

### Checking Expiration

```swift
extension Invoice {
    var isExpired: Bool {
        return Date().timeIntervalSince1970 > TimeInterval(expiry)
    }
    
    var timeUntilExpiry: TimeInterval {
        return TimeInterval(expiry) - Date().timeIntervalSince1970
    }
    
    var expiryDate: Date {
        return Date(timeIntervalSince1970: TimeInterval(expiry))
    }
}

// Usage
if invoice.isExpired {
    showError("Invoice has expired")
} else {
    let remaining = invoice.timeUntilExpiry
    print("Expires in \(Int(remaining / 60)) minutes")
}
```

## Payment Hash

The payment hash is used to identify and verify Lightning payments:

```swift
extension Invoice {
    var paymentHashHex: String {
        return paymentHash.map { String(format: "%02x", $0) }.joined()
    }
    
    func verifyPreimage(_ preimage: Data) -> Bool {
        let hash = SHA256.hash(data: preimage)
        return Data(hash) == paymentHash
    }
}

// Usage
print("Payment hash: \(invoice.paymentHashHex)")

if invoice.verifyPreimage(receivedPreimage) {
    print("Preimage is valid!")
}
```

### iOS Helper Extensions

```swift
extension MuunPaymentURI {
    var hasLightningOption: Bool {
        return invoice != nil
    }
    
    var hasOnChainOption: Bool {
        return !address.isEmpty
    }
    
    var isUnifiedQR: Bool {
        return hasLightningOption && hasOnChainOption
    }
    
    var preferredPaymentMethod: PaymentMethod {
        // Prefer Lightning if available and not expired
        if let invoice = invoice, !invoice.isExpired {
            return .lightning
        }
        return .onChain
    }
}

enum PaymentMethod {
    case onChain
    case lightning
}
```

## Error Handling

### Invoice Errors

```swift
enum InvoiceError: Error {
    case invalidFormat
    case invalidNetwork
    case expired
    case amountMismatch
    case invalidScheme
    
    var localizedDescription: String {
        switch self {
        case .invalidFormat:
            return "The invoice format is invalid"
        case .invalidNetwork:
            return "This invoice is for a different network"
        case .expired:
            return "This invoice has expired"
        case .amountMismatch:
            return "On-chain and Lightning amounts don't match"
        case .invalidScheme:
            return "Invalid URI scheme"
        }
    }
}

func parseInvoiceSafe(_ invoiceString: String) -> Result<Invoice, InvoiceError> {
    do {
        let invoice = try LibwalletParseInvoice(invoiceString, network)
        
        if invoice.isExpired {
            return .failure(.expired)
        }
        
        return .success(invoice)
    } catch let error as NSError {
        if error.domain.contains("invalid_invoice") {
            return .failure(.invalidFormat)
        }
        return .failure(.invalidFormat)
    }
}
```
