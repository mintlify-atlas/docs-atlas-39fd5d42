## Overview

Submarine swaps allow converting Lightning payments to on-chain Bitcoin outputs. Muun supports two submarine swap versions for sending and receiving Lightning payments.

## Submarine Swap V1

### V1 Witness Script

V1 swaps use time-locked refund scripts:

```go
func createWitnessScriptV1(
    userRefundAddress string,
    swapServerPublicKey []byte,
    paymentHash256 []byte,
    numBlocksForExpiration int64,
    network *chaincfg.Params,
) ([]byte, error) {
    // Script: 
    // OP_DUP OP_HASH160 <swap_server_pubkey_hash> OP_EQUAL
    // OP_IF
    //     OP_HASH160 <payment_hash> OP_EQUALVERIFY
    // OP_ELSE
    //     <num_blocks> OP_CHECKSEQUENCEVERIFY OP_DROP
    //     OP_DUP OP_HASH160 <user_refund_address_hash> OP_EQUALVERIFY
    // OP_ENDIF
    // OP_CHECKSIG
    
    builder := txscript.NewScriptBuilder()
    
    // If branch (swap server with preimage)
    builder.AddOp(txscript.OP_DUP)
    builder.AddOp(txscript.OP_HASH160)
    builder.AddData(btcutil.Hash160(swapServerPublicKey))
    builder.AddOp(txscript.OP_EQUAL)
    builder.AddOp(txscript.OP_IF)
    builder.AddOp(txscript.OP_HASH160)
    builder.AddData(paymentHash256)
    builder.AddOp(txscript.OP_EQUALVERIFY)
    
    // Else branch (user refund after timeout)
    builder.AddOp(txscript.OP_ELSE)
    builder.AddInt64(numBlocksForExpiration)
    builder.AddOp(txscript.OP_CHECKSEQUENCEVERIFY)
    builder.AddOp(txscript.OP_DROP)
    builder.AddOp(txscript.OP_DUP)
    builder.AddOp(txscript.OP_HASH160)
    
    refundAddr, err := btcutilw.DecodeAddress(userRefundAddress, network)
    if err != nil {
        return nil, err
    }
    builder.AddData(refundAddr.ScriptAddress())
    builder.AddOp(txscript.OP_EQUALVERIFY)
    builder.AddOp(txscript.OP_ENDIF)
    builder.AddOp(txscript.OP_CHECKSIG)
    
    return builder.Script()
}
```

```go
func createNonNativeSegwitRedeemScript(witnessScript []byte) ([]byte, error) {
    witnessScriptHash := sha256.Sum256(witnessScript)
    
    builder := txscript.NewScriptBuilder()
    builder.AddInt64(0)
    builder.AddData(witnessScriptHash[:])
    
    return builder.Script()
}
```

## Submarine Swap V2

### V2 Features

- Uses MuSig for user/Muun keys
- More efficient witness scripts
- Better privacy
- Lower fees

### V2 Witness Script

```go
func createWitnessScriptV2(
    userPublicKey []byte,
    muunPublicKey []byte,
    swapServerPublicKey []byte,
    paymentHash256 []byte,
    numBlocksForExpiration int64,
) ([]byte, error) {
    // Combine user and muun keys with MuSig
    pubKeys := [][]byte{userPublicKey, muunPublicKey}
    tweak := musig.KeySpendOnlyTweak()
    
    aggregateKey, err := musig.Musig2CombinePubKeysWithTweak(
        musig.Musig2v040Muun,
        pubKeys,
        tweak,
    )
    if err != nil {
        return nil, err
    }
    
    aggregatedPublicKey := aggregateKey.FinalKey.SerializeCompressed()
    
    // Build script:
    // <aggregated_pubkey> OP_CHECKSIGVERIFY OP_HASH160 <payment_hash> OP_EQUAL
    // OP_IF
    //     <swap_server_pubkey>
    // OP_ELSE
    //     <num_blocks> OP_CHECKSEQUENCEVERIFY OP_DROP
    //     <aggregated_pubkey>
    // OP_ENDIF
    // OP_CHECKSIG
    
    builder := txscript.NewScriptBuilder()
    builder.AddData(aggregatedPublicKey)
    builder.AddOp(txscript.OP_CHECKSIGVERIFY)
    builder.AddOp(txscript.OP_HASH160)
    builder.AddData(ripemd160(paymentHash256))
    builder.AddOp(txscript.OP_EQUAL)
    builder.AddOp(txscript.OP_IF)
    builder.AddData(swapServerPublicKey)
    builder.AddOp(txscript.OP_ELSE)
    builder.AddInt64(numBlocksForExpiration)
    builder.AddOp(txscript.OP_CHECKSEQUENCEVERIFY)
    builder.AddOp(txscript.OP_DROP)
    builder.AddData(aggregatedPublicKey)
    builder.AddOp(txscript.OP_ENDIF)
    builder.AddOp(txscript.OP_CHECKSIG)
    
    return builder.Script()
}

func ripemd160(data []byte) []byte {
    hasher := hash.New()
    _, err := hasher.Write(data)
    if err != nil {
        panic("failed to hash")
    }
    return hasher.Sum([]byte{})
}
```

```go Go (libwallet/swaps/swaps.go:79)
func (swap *SubmarineSwap) Validate(
    rawInvoice string,
    userPublicKey *KeyDescriptor,
    muunPublicKey *KeyDescriptor,
    originalExpirationInBlocks int64,
    network *chaincfg.Params,
) error {
    version := swap.FundingOutput.ScriptVersion
    switch version {
    case addresses.SubmarineSwapV1:
        return swap.validateV1(
            rawInvoice,
            userPublicKey,
            muunPublicKey,
            network,
        )
    case addresses.SubmarineSwapV2:
        return swap.validateV2(
            rawInvoice,
            userPublicKey,
            muunPublicKey,
            originalExpirationInBlocks,
            network,
        )
    default:
        return fmt.Errorf("unknown swap version %v", version)
    }
}
```

## Key Descriptor

### KeyDescriptor Structure

```go
type KeyDescriptor struct {
    Key  *hdkeychain.ExtendedKey
    Path string
}

func (d *KeyDescriptor) DeriveTo(path string) (*hdkeychain.ExtendedKey, error) {
    key := d.Key
    
    currentPath, err := hdpath.Parse(d.Path)
    if err != nil {
        return nil, fmt.Errorf("invalid current key path: %w", err)
    }
    targetPath, err := hdpath.Parse(path)
    if err != nil {
        return nil, fmt.Errorf("invalid target key path: %w", err)
    }
    indexes := targetPath.IndexesFrom(currentPath)
    for _, index := range indexes {
        var err error
        var modifier uint32
        if index.Hardened {
            modifier = hdkeychain.HardenedKeyStart
        }
        key, err = key.Derive(index.Index | modifier)
        if err != nil {
            return nil, err
        }
    }
    return key, nil
}
```

## Spending Swap Outputs

### With Preimage (Success)

```go
func spendSwapWithPreimage(
    tx *wire.MsgTx,
    index int,
    witnessScript []byte,
    preimage []byte,
    userSignature []byte,
    muunSignature []byte,
) error {
    // V2: User+Muun signature, preimage, script
    witness := [][]byte{
        userSignature,
        muunSignature,
        preimage,
        []byte{1}, // Choose IF branch
        witnessScript,
    }
    
    tx.TxIn[index].Witness = witness
    return nil
}
```

## Swap Coins

### CoinSubmarineSwapV1

```go
type coinSubmarineSwapV1 struct {
    Network         *chaincfg.Params
    OutPoint        wire.OutPoint
    KeyPath         string
    Amount          btcutil.Amount
    RefundAddress   string
    PaymentHash256  []byte
    ServerPublicKey []byte
    LockTime        int64
}

func (c *coinSubmarineSwapV1) SignInput(
    index int,
    tx *wire.MsgTx,
    userKey *HDPrivateKey,
    muunKey *HDPublicKey,
) error {
    // Derive keys
    derivedUserKey, err := userKey.DeriveTo(c.KeyPath)
    if err != nil {
        return err
    }
    
    // Build witness script
    witnessScript, err := createWitnessScriptV1(
        c.RefundAddress,
        c.ServerPublicKey,
        c.PaymentHash256,
        c.LockTime,
        c.Network,
    )
    if err != nil {
        return err
    }
    
    // Sign for refund path
    signature, err := c.signForRefund(
        tx,
        index,
        witnessScript,
        derivedUserKey,
    )
    if err != nil {
        return err
    }
    
    // Build witness
    witness := [][]byte{
        signature,
        []byte{},  // Empty preimage (refund)
        []byte{0}, // Choose ELSE branch
        witnessScript,
    }
    
    tx.TxIn[index].Witness = witness
    return nil
}
```

### CoinSubmarineSwapV2

```go
type coinSubmarineSwapV2 struct {
    Network             *chaincfg.Params
    OutPoint            wire.OutPoint
    KeyPath             string
    Amount              btcutil.Amount
    PaymentHash256      []byte
    UserPublicKey       []byte
    MuunPublicKey       []byte
    ServerPublicKey     []byte
    BlocksForExpiration int64
    ServerSignature     []byte
}

func (c *coinSubmarineSwapV2) SignInput(
    index int,
    tx *wire.MsgTx,
    userKey *HDPrivateKey,
    muunKey *HDPublicKey,
) error {
    // Derive keys
    derivedUserKey, err := userKey.DeriveTo(c.KeyPath)
    derivedMuunKey, err := muunKey.DeriveTo(c.KeyPath)
    
    // Build witness script
    witnessScript, err := createWitnessScriptV2(
        c.UserPublicKey,
        c.MuunPublicKey,
        c.ServerPublicKey,
        c.PaymentHash256,
        c.BlocksForExpiration,
    )
    if err != nil {
        return err
    }
    
    // Sign with MuSig
    userSignature, err := c.signWithMusig(
        tx,
        index,
        witnessScript,
        derivedUserKey,
        derivedMuunKey,
    )
    if err != nil {
        return err
    }
    
    // Combine signatures and build witness
    tx.TxIn[index].Witness = c.buildWitness(
        userSignature,
        c.ServerSignature,
        witnessScript,
    )
    
    return nil
}
```

## Complete Swap Flow

### iOS Implementation

```swift
class SubmarineSwapManager {
    func createSwap(
        invoice: String,
        amount: Int64
    ) async throws -> SubmarineSwap {
        // 1. Request swap from server
        let swapRequest = SwapRequest(
            invoice: invoice,
            amount: amount
        )
        
        let swapResponse = try await apiClient.createSwap(swapRequest)
        
        // 2. Validate swap parameters
        let swap = try LibwalletSubmarineSwap(from: swapResponse)
        
        try swap.validate(
            invoice,
            userPublicKey: userKey,
            muunPublicKey: muunKey,
            originalExpirationInBlocks: 144,
            network: network
        )
        
        // 3. Create funding transaction
        let fundingTx = try createFundingTransaction(
            outputAddress: swap.fundingOutput.outputAddress,
            amount: swap.fundingOutput.outputAmount
        )
        
        // 4. Sign and broadcast
        let signedTx = try fundingTx.sign(userPrivKey, muunKey)
        try await broadcast(signedTx)
        
        return swap
    }
    
    func refundExpiredSwap(
        _ swap: SubmarineSwap
    ) async throws {
        // Check if swap can be refunded
        guard swap.canRefund else {
            throw SwapError.cannotRefund
        }
        
        // Create refund transaction
        let refundTx = try createRefundTransaction(swap)
        
        // Sign with both keys
        let signedTx = try refundTx.fullySign(userPrivKey, muunPrivKey)
        
        // Broadcast refund
        try await broadcast(signedTx)
    }
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Validate Before Funding">
    Always validate swap parameters before sending funds:
    ```swift
    do {
        try swap.validate(
            invoice,
            userPublicKey: userKey,
            muunPublicKey: muunKey,
            originalExpirationInBlocks: expectedBlocks,
            network: network
        )
    } catch {
        // Don't fund invalid swaps
        throw SwapError.invalidParameters
    }
    ```
  </Accordion>
  
  <Accordion title="Monitor Expiration">
    Track swap expiration to enable refunds:
    ```swift
    func monitorSwapExpiration(_ swap: SubmarineSwap) {
        let expiryBlock = swap.fundingOutput.expirationInBlocks
        
        // Poll current block height
        Task {
            while true {
                let currentHeight = try await getBlockHeight()
                
                if currentHeight >= expiryBlock {
                    // Swap can be refunded
                    try await refundExpiredSwap(swap)
                    break
                }
                
                try await Task.sleep(nanoseconds: 60_000_000_000) // 1 min
            }
        }
    }
    ```
  </Accordion>

## Error Handling

```swift
enum SwapError: Error {
    case invalidVersion
    case expiredInvoice
    case amountMismatch
    case invalidScript
    case cannotRefund
    case insufficientConfirmations
    
    var localizedDescription: String {
        switch self {
        case .invalidVersion:
            return "Unsupported swap version"
        case .expiredInvoice:
            return "Lightning invoice has expired"
        case .amountMismatch:
            return "Swap amount doesn't match invoice"
        case .invalidScript:
            return "Invalid swap script"
        case .cannotRefund:
            return "Swap cannot be refunded yet"
        case .insufficientConfirmations:
            return "Waiting for confirmations"
        }
    }
}
```
