## Overview

Libwallet provides encryption utilities using AES-256-CBC with ECDHE (Elliptic Curve Diffie-Hellman Ephemeral) for secure key exchange and data encryption.

## Encryption Methods

### AES-CBC

Libwallet uses AES in CBC (Cipher Block Chaining) mode:

```go
const KeySize = 32 // 256 bits

func EncryptPkcs7(
    key []byte,
    iv []byte,
    plaintext []byte,
) ([]byte, error) {
    plaintext = pkcs7Padding(plaintext)
    return EncryptNoPadding(key, iv, plaintext)
}

func EncryptNoPadding(
    key []byte,
    iv []byte,
    plaintext []byte,
) ([]byte, error) {
    if len(key) != KeySize {
        return nil, fmt.Errorf("invalid key size, expected %v, got %v",
            KeySize, len(key))
    }
    
    block, err := aes.NewCipher(key)
    if err != nil {
        return nil, err
    }
    
    ciphertext := make([]byte, len(plaintext))
    mode := cipher.NewCBCEncrypter(block, iv)
    mode.CryptBlocks(ciphertext, plaintext)
    
    return ciphertext, nil
}
```

### PKCS7 Padding

```go
func pkcs7Padding(src []byte) []byte {
    padding := aes.BlockSize - len(src)%aes.BlockSize
    padtext := bytes.Repeat([]byte{byte(padding)}, padding)
    return append(src, padtext...)
}

func pkcs7UnPadding(src []byte) ([]byte, error) {
    length := len(src)
    unpadding := int(src[length-1])
    
    if unpadding > aes.BlockSize || unpadding == 0 {
        return nil, errors.New("invalid pkcs7 padding")
    }
    
    pad := src[len(src)-unpadding:]
    for i := 0; i < unpadding; i++ {
        if pad[i] != byte(unpadding) {
            return nil, errors.New("invalid pkcs7 padding")
        }
    }
    
    return src[:(length - unpadding)], nil
}
```

## ECDHE Encryption

### Encrypt with Public Key

<CodeGroup>

```go Go (libwallet/encrypt.go:31)
func encryptWithPubKey(
    pubKey *btcec.PublicKey,
    plaintext []byte,
) (*btcec.PublicKey, []byte, error) {
    // Generate ephemeral key pair
    pubEph, sharedSecret, err := encryption.GenerateSharedEncryptionSecret(pubKey)
    if err != nil {
        return nil, nil, err
    }
    
    // Use last 16 bytes of ephemeral pubkey as IV
    serializedPubkey := pubEph.SerializeCompressed()
    iv := serializedPubkey[len(serializedPubkey)-aes.BlockSize:]
    
    // Encrypt with shared secret
    ciphertext, err := aescbc.EncryptNoPadding(sharedSecret, iv, plaintext)
    if err != nil {
        return nil, nil, fmt.Errorf("encryptWithPubKey: encrypt failed: %w", err)
    }
    
    return pubEph, ciphertext, nil
}
```

### Decrypt with Private Key

```go
func decryptWithPrivKey(
    privKey *btcec.PrivateKey,
    rawPubEph []byte,
    ciphertext []byte,
) ([]byte, error) {
    // Recover shared secret from ephemeral pubkey
    sharedSecret, err := encryption.RecoverSharedEncryptionSecret(
        privKey,
        rawPubEph,
    )
    if err != nil {
        return nil, err
    }
    
    // Extract IV from ephemeral pubkey
    iv := rawPubEph[len(rawPubEph)-aes.BlockSize:]
    
    // Decrypt
    plaintext, err := aescbc.DecryptNoPadding(sharedSecret, iv, ciphertext)
    if err != nil {
        return nil, fmt.Errorf("decryptWithPrivKey: failed to decrypt: %w", err)
    }
    
    return plaintext, nil
}
```

## Shared Secret Generation

### ECDH Shared Secret

```go
func GenerateSharedEncryptionSecret(
    recipientPubKey *btcec.PublicKey,
) (*btcec.PublicKey, []byte, error) {
    // Generate random ephemeral private key
    ephemeralPrivKey, err := btcec.NewPrivateKey()
    if err != nil {
        return nil, nil, err
    }
    
    // Compute shared secret: ephemeralPriv * recipientPub
    sharedSecret := btcec.GenerateSharedSecret(
        ephemeralPrivKey,
        recipientPubKey,
    )
    
    // Hash the shared point to get 32-byte key
    key := sha256.Sum256(sharedSecret)
    
    return ephemeralPrivKey.PubKey(), key[:], nil
}

func RecoverSharedEncryptionSecret(
    recipientPrivKey *btcec.PrivateKey,
    ephemeralPubKeyBytes []byte,
) ([]byte, error) {
    // Parse ephemeral public key
    ephemeralPubKey, err := btcec.ParsePubKey(ephemeralPubKeyBytes)
    if err != nil {
        return nil, err
    }
    
    // Compute shared secret: recipientPriv * ephemeralPub
    sharedSecret := btcec.GenerateSharedSecret(
        recipientPrivKey,
        ephemeralPubKey,
    )
    
    // Hash to get same 32-byte key
    key := sha256.Sum256(sharedSecret)
    
    return key[:], nil
}
```

## Encrypted Payload Format

### Payload Structure

```
+------------------+------------------+
| Ephemeral PubKey | Ciphertext       |
| (33 bytes)       | (variable)       |
+------------------+------------------+
```

### Encoding/Decoding

```swift
struct EncryptedPayload {
    let ephemeralPubKey: Data  // 33 bytes compressed
    let ciphertext: Data
    
    func encode() -> String {
        var combined = Data()
        combined.append(ephemeralPubKey)
        combined.append(ciphertext)
        return combined.base64EncodedString()
    }
    
    static func decode(_ encoded: String) throws -> EncryptedPayload {
        guard let data = Data(base64Encoded: encoded) else {
            throw EncryptionError.invalidEncoding
        }
        
        guard data.count > 33 else {
            throw EncryptionError.invalidPayload
        }
        
        let ephemeralPubKey = data.prefix(33)
        let ciphertext = data.dropFirst(33)
        
        return EncryptedPayload(
            ephemeralPubKey: ephemeralPubKey,
            ciphertext: ciphertext
        )
    }
}
```

## Key Derivation

### Derive Encryption Key from Password

```swift
import CryptoKit

func deriveEncryptionKey(
    password: String,
    salt: Data,
    iterations: Int = 100_000
) -> SymmetricKey {
    let passwordData = password.data(using: .utf8)!
    
    return PBKDF2.deriveKey(
        password: passwordData,
        salt: salt,
        iterations: iterations,
        keyLength: 32
    )
}

// Usage
let password = "user_password"
let salt = Data.random(count: 32)
let key = deriveEncryptionKey(password: password, salt: salt)

// Store salt for future decryption
UserDefaults.standard.set(salt, forKey: "encryption_salt")
```

## Best Practices

<AccordionGroup>
  <Accordion title="Always Use Random IVs">
    Never reuse IVs with the same key:
    ```swift
    // ✅ CORRECT - unique IV per encryption
    func encrypt(_ data: Data, key: Data) throws -> (Data, Data) {
        let iv = Data.random(count: 16)
        let ciphertext = try AES.encrypt(data, key: key, iv: iv)
        return (ciphertext, iv)
    }
    
    // ❌ WRONG - fixed IV is insecure
    let fixedIV = Data(repeating: 0, count: 16)
    let ciphertext = try AES.encrypt(data, key: key, iv: fixedIV)
    ```
  </Accordion>
  
  <Accordion title="Validate Key Sizes">
    Always validate key and IV lengths:
    ```swift
    func validateEncryptionParams(
        key: Data,
        iv: Data
    ) throws {
        guard key.count == 32 else {
            throw EncryptionError.invalidKeySize
        }
        
        guard iv.count == 16 else {
            throw EncryptionError.invalidIVSize
        }
    }
    ```
  </Accordion>
  
  <Accordion title="Clear Sensitive Data">
    Explicitly zero out sensitive data:
    ```swift
    func encryptSensitiveData(_ data: Data, key: Data) throws -> Data {
        var mutableData = data
        var mutableKey = key
        
        defer {
            // Zero out sensitive data
            mutableData.resetBytes(in: 0..<mutableData.count)
            mutableKey.resetBytes(in: 0..<mutableKey.count)
        }
        
        return try performEncryption(mutableData, key: mutableKey)
    }
    ```
  </Accordion>
  
  <Accordion title="Use Authenticated Encryption">
    Consider adding HMAC for authentication:
    ```swift
    func encryptAndAuthenticate(
        _ plaintext: Data,
        encryptionKey: Data,
        macKey: Data
    ) throws -> Data {
        // Encrypt
        let iv = Data.random(count: 16)
        let ciphertext = try AES.encrypt(
            plaintext,
            key: encryptionKey,
            iv: iv
        )
        
        // Compute MAC over IV || ciphertext
        var authenticated = iv
        authenticated.append(ciphertext)
        let mac = HMAC<SHA256>.authenticationCode(
            for: authenticated,
            using: SymmetricKey(data: macKey)
        )
        
        // Return IV || ciphertext || MAC
        authenticated.append(contentsOf: mac)
        return authenticated
    }
    ```
  </Accordion>
</AccordionGroup>

## Error Handling

```swift
enum EncryptionError: Error {
    case invalidKeySize
    case invalidIVSize
    case invalidPadding
    case invalidEncoding
    case invalidPayload
    case encryptionFailed
    case decryptionFailed
    
    var localizedDescription: String {
        switch self {
        case .invalidKeySize:
            return "Encryption key must be 32 bytes"
        case .invalidIVSize:
            return "IV must be 16 bytes"
        case .invalidPadding:
            return "Invalid PKCS7 padding"
        case .invalidEncoding:
            return "Failed to decode encrypted payload"
        case .invalidPayload:
            return "Encrypted payload is too short"
        case .encryptionFailed:
            return "Encryption operation failed"
        case .decryptionFailed:
            return "Decryption operation failed"
        }
    }
}
```

## Complete Example

### iOS Encryption Manager

```swift
class EncryptionManager {
    private let privateKey: LibwalletHDPrivateKey
    
    init(privateKey: LibwalletHDPrivateKey) {
        self.privateKey = privateKey
    }
    
    // Encrypt for recipient
    func encrypt(
        _ message: String,
        for recipientPublicKey: LibwalletHDPublicKey
    ) throws -> String {
        let messageData = message.data(using: .utf8)!
        
        let operation = LibwalletNewEncryptOperation(
            recipientPublicKey,
            messageData
        )
        
        return try operation.encrypt()
    }
    
    // Decrypt from sender
    func decrypt(
        _ encryptedPayload: String,
        from senderPublicKey: LibwalletPublicKey? = nil
    ) throws -> String {
        let operation: LibwalletDecryptOperation
        
        if let sender = senderPublicKey {
            operation = LibwalletNewDecryptOperationFrom(
                sender,
                privateKey,
                encryptedPayload
            )
        } else {
            operation = LibwalletNewDecryptOperation(
                privateKey,
                encryptedPayload
            )
        }
        
        let decrypted = try operation.decrypt()
        
        guard let message = String(data: decrypted, encoding: .utf8) else {
            throw EncryptionError.decryptionFailed
        }
        
        return message
    }
    
    // Encrypt sensitive local data
    func encryptLocal(_ data: Data, password: String) throws -> Data {
        let salt = Data.random(count: 32)
        let key = deriveEncryptionKey(
            password: password,
            salt: salt
        )
        
        let iv = Data.random(count: 16)
        let ciphertext = try AES.encrypt(data, key: Data(key), iv: iv)
        
        // Return salt || IV || ciphertext
        var result = salt
        result.append(iv)
        result.append(ciphertext)
        return result
    }
    
    // Decrypt local data
    func decryptLocal(_ encrypted: Data, password: String) throws -> Data {
        guard encrypted.count > 48 else { // 32 salt + 16 IV
            throw EncryptionError.invalidPayload
        }
        
        let salt = encrypted.prefix(32)
        let iv = encrypted.dropFirst(32).prefix(16)
        let ciphertext = encrypted.dropFirst(48)
        
        let key = deriveEncryptionKey(
            password: password,
            salt: salt
        )
        
        return try AES.decrypt(ciphertext, key: Data(key), iv: iv)
    }
}
```
