## Overview

Domain models represent the core business entities in the Muun Wallet. These are pure Swift structs and enums that encapsulate business logic and are independent of any framework or data source.

#### Business Logic Methods

```swift
// Check if operation can be canceled (RBF)
public func isCancelable() -> Bool {
    return transaction?.isReplaceableByFee ?? false
        && isPending()
}

// Check if operation is pending
public func isPending() -> Bool {
    return OperationStatus.pendingStates.contains(status)
}

// Calculate total fees (on-chain + off-chain)
public func totalFeeInSatoshis() -> Satoshis {
    return fee.inSatoshis + offChainFeeInSatoshis()
}

private func offChainFeeInSatoshis() -> Satoshis {
    if let swap = submarineSwap {
        if let debtType = swap._fundingOutput._debtType, debtType == .LEND {
            return swap._fees?._lightning ?? Satoshis(value: 0)
        } else {
            return swap._fees?.total() ?? Satoshis(value: 0)
        }
    }
    return Satoshis(value: 0)
}

// Check if operation failed and was outgoing
public func isFailedAndOutgoing() -> Bool {
    return status == .FAILED && direction == .OUTGOING
}
```

### OperationStatus

```swift
public enum OperationStatus: String, Codable {
    case CREATED                 // Newly created, unsigned
    case SIGNING                 // Being signed
    case SIGNED                  // Signed but not broadcasted
    case BROADCASTED             // Broadcasted, awaiting confirmation
    case SWAP_PENDING            // Submarine swap: on-chain confirmed, awaiting off-chain
    case SWAP_OPENING_CHANNEL    // Opening Lightning channel
    case SWAP_WAITING_CHANNEL    // Waiting for channel to open
    case SWAP_ROUTING            // Routing Lightning payment
    case SWAP_PAYED              // Lightning payment successful
    case SWAP_FAILED             // Lightning payment failed
    case SWAP_EXPIRED            // Swap expired, funds time-locked
    case CONFIRMED               // Confirmed but not settled
    case SETTLED                 // Fully confirmed (6+ confirmations)
    case DROPPED                 // Not found in recent blockchain sync
    case FAILED                  // Rejected by network
    
    static let pendingStates: [OperationStatus] = [
        .CREATED, .SIGNING, .SIGNED, .BROADCASTED,
        .SWAP_PENDING, .SWAP_OPENING_CHANNEL,
        .SWAP_WAITING_CHANNEL, .SWAP_ROUTING
    ]
}
```

## Common Patterns

### Immutable Structs

Most models are immutable structs:

```swift
public struct Operation {
    public let id: Int?
    public let direction: OperationDirection
    // Properties are let (immutable)
    
    // Create new instance for updates
    func withStatus(_ newStatus: OperationStatus) -> Operation {
        return Operation(
            id: self.id,
            direction: self.direction,
            status: newStatus,
            // ... copy other properties
        )
    }
}
```

### Business Logic in Models

Models contain domain logic:

```swift
public struct Operation {
    // Business rules encoded in the model
    public func isCancelable() -> Bool {
        return transaction?.isReplaceableByFee ?? false && isPending()
    }
    
    public func isPending() -> Bool {
        return OperationStatus.pendingStates.contains(status)
    }
    
    public func totalFeeInSatoshis() -> Satoshis {
        return fee.inSatoshis + offChainFeeInSatoshis()
    }
}
```

### Codable for Serialization

All models conform to `Codable`:

```swift
public struct User: Codable {
    public let id: Int
    let firstName: String?
    // Automatically encoded/decoded to/from JSON
}

// Usage
let encoder = JSONEncoder()
let data = try encoder.encode(user)

let decoder = JSONDecoder()
let user = try decoder.decode(User.self, from: data)
```

### Enums with Raw Values

```swift
public enum OperationStatus: String, Codable {
    case CREATED
    case BROADCASTED

// From database
```

### Database to Domain


### DTO to Domain

```swift
// DTO from API
struct OperationJson: Codable {
    let id: Int
    let status: String
    let amount: AmountJson
    
    func toModel() -> Operation {
        return Operation(
            id: id,
            status: OperationStatus(rawValue: status)!,
            amount: amount.toModel()
        )
    }
}

struct AmountJson: Codable {
    let inSatoshis: Int
    let inInputCurrency: MonetaryAmountJson
    
    func toModel() -> BitcoinAmount {
        return BitcoinAmount(
            inSatoshis: Satoshis(value: inSatoshis),
            inInputCurrency: inInputCurrency.toModel()
        )
    }
}
```

## Model Guidelines

<AccordionGroup>
  <Accordion title="Pure Swift">
    Domain models should not depend on UIKit, Foundation (except basic types), or any framework. This makes them portable and testable.
  </Accordion>

  <Accordion title="Immutability">
    Prefer immutable `let` properties. Use `copy()` methods or `with*()` methods for updates.
  </Accordion>

  <Accordion title="Value Semantics">
    Use `struct` instead of `class` to get value semantics (copy on mutation).
  </Accordion>

  <Accordion title="Business Logic">
    Encapsulate domain rules in the model. For example, `isCancelable()` instead of checking conditions in multiple places.
  </Accordion>

  <Accordion title="Codable">
    Implement `Codable` for serialization to JSON/database. Use custom coding keys if needed.
  </Accordion>

  <Accordion title="Optionals">
    Use optionals (`?`) for nullable fields, not sentinel values like -1 or empty strings.
  </Accordion>
</AccordionGroup>

## Example Usage

### In Repository

```swift
class OperationRepository: BaseDatabaseRepository<OperationDB, Operation> {
    
    func getPendingOperations() -> [Operation] {
        let query = OperationDB
            .filter(OperationStatus.pendingStates.map { $0.rawValue }
                .contains(Column("status")))
        
        return objects(query: query) ?? []
    }
}
```

### In Use Case

```swift
class CancelOperationAction: AsyncAction<Operation> {
    
    func run(operationId: Int) {
        let single = Single.deferred { [weak self] in
            guard let operation = self?.operationRepository.object(with: operationId),
                  operation.isCancelable() else {
                throw OperationError.notCancelable
            }
            
            return self?.houstonService.cancelOperation(operationId) ?? .error(MuunError())
        }
        
        runSingle(single)
    }
}
```

### In View Controller

```swift
class OperationDetailViewController: MUViewController {
    
    let operation: Operation
    
    override func viewDidLoad() {
        super.viewDidLoad()

        // Business logic in model

        // Computed properties
        feeLabel.text = formatBitcoin(operation.totalFeeInSatoshis())
    }
}
```
