## Overview

Domain services encapsulate complex business logic that doesn't belong in a single model or action. They coordinate operations, perform calculations, and provide reusable domain logic.

### Automatic Retry

```swift
func shouldRetry(error: Error) -> Bool {
    if let muunError = error as? MuunError,
       let e = muunError.kind as? ServiceError {
        switch e {
        case .serviceFailure, .timeOut:
            return true
        default:
            return false
        }
    }
    return false
}

// Retry logic in performHTTPRequest
if self.shouldRetry(error: someError) && maxRetries > 1 {

    self.performHTTPRequest(
        request: newRequest,
        model: model,
        maxRetries: maxRetries - 1,
        success: success,
        failure: failure
    )
}
```

### Error Parsing

```swift
func parseError(data: Data?, error: Error?, response: URLResponse?) -> Error? {
    // 5xx errors
    if let httpResponse = response as? HTTPURLResponse {
        if httpResponse.statusCode >= 500 {
            return MuunError(ServiceError.serviceFailure)
        }
    }
    
    // Network errors
    if let e = error as? URLError {
        switch e.code {
        case .timedOut:
            return MuunError(ServiceError.timeOut)
            
        case .cannotFindHost,
             .cannotConnectToHost,
             .notConnectedToInternet,
             .networkConnectionLost:
            return MuunError(ServiceError.internetError)
            
        default:
            return MuunError(e)
        }
    }
    
    // Custom API errors
    guard let data = data else { return nil }
    
    let jsonDecoder = JSONDecoder()
    do {
        let error = try jsonDecoder.decode(DeveloperError.self, from: data)
        return MuunError(ServiceError.customError(error))
    } catch {
        return nil
    }
}
```

## Houston Service

The main API service for communicating with Muun backend:

```swift
class HoustonService: BaseService {
    
    override func getBaseURL() -> String {
        return Environment.current.houstonURL
    }
    
    // User operations
    func fetchUser() -> Single<User>
    func updateUser(_ user: User) -> Single<User>
    
    // Operations
    func fetchOperations() -> Single<[Operation]>
    func createOperation(amount: Satoshis, address: String, fee: Satoshis) -> Single<Operation>
    func submitOperation(_ operation: Operation) -> Single<Operation>
    
    // Swaps
    func createSubmarineSwap(invoice: String) -> Single<SubmarineSwap>
    func fulfillIncomingSwap(uuid: String) -> Completable
    
    // Real-time data
    func fetchExchangeRates() -> Single<ExchangeRateWindow>
    func fetchFees() -> Single<FeeWindow>
    func fetchBlockchainHeight() -> Single<Int>
    
    // Authentication
    func login(email: String, password: String) -> Single<LoginResponse>
    func createSession(gcmToken: String) -> Single<Session>
}
```

### Example Usage

```swift
class FetchOperationsAction: AsyncAction<[Operation]> {
    private let houstonService: HoustonService
    private let operationRepository: OperationRepository
    
    func run() {
        let single = houstonService.fetchOperations()
            .do(onSuccess: { [weak self] operations in
                _ = self?.operationRepository
                    .storeOperations(operations)
                    .subscribe()
            })
        
        runSingle(single)
    }
}
```

### Example Usage

```swift
class SignTransactionAction: AsyncAction<Operation> {
    private let walletService: WalletService
    private let keysRepository: KeysRepository
    
    func run(operation: Operation) {
        let single = Single.deferred { [weak self] in
            guard let self = self,
                  let transaction = operation.transaction else {
                throw TransactionError.invalid
            }
            
            let privateKey = try self.keysRepository.getPrivateKey()
            let wallet = try self.keysRepository.getHDWallet()
            
            let signedTx = try self.walletService.signTransaction(
                wallet: wallet,
                transaction: transaction,
                privateKey: privateKey
            )
            
            var signed = operation
            signed.transaction?.hash = signedTx
            signed.status = .SIGNED
            
            return Single.just(signed)
        }
        
        runSingle(single)
    }
}
```

### FeatureFlagsSelector

```swift
class FeatureFlagsSelector {
    private let featureFlagsRepository: FeatureFlagsRepository
    private let userRepository: UserRepository
    
    func isFeatureEnabled(_ feature: Feature) -> Bool {
        let flags = featureFlagsRepository.getFlags()
        let user = userRepository.getUser()
        
        // Apply flag logic
        return evaluateFlag(feature, flags: flags, user: user)
    }
    
    func watchFeatureFlag(_ feature: Feature) -> Observable<Bool> {
        return Observable.combineLatest(
            featureFlagsRepository.watchFlags(),
            userRepository.watchUser()
        )
        .map { (flags, user) in
            return self.evaluateFlag(feature, flags: flags, user: user)
        }
    }
}
```

### ExportKeysSelector

```swift
class ExportKeysSelector {
    private let keysRepository: KeysRepository
    private let userRepository: UserRepository
    
    func canExportKeys() -> Bool {
        guard let user = userRepository.getUser() else {
            return false
        }
        
        return user.hasPasswordChallengeKey && !user.hasExportedKeys
    }
    
    func getExportableKeys() -> [String: String] {
        return [
            "private_key": try! keysRepository.getPrivateKey(),
            "base_path": try! keysRepository.getBaseDerivationPath()
        ]
    }
}
```

```swift
class FeeService {
    private let houstonService: HoustonService
    private let feeRepository: FeeWindowRepository
    
    func watchFees() -> Observable<FeeWindow> {
        // Watch local fees
        return feeRepository.watchFeeWindow()
    }
    
    func refreshFees() -> Completable {
        return houstonService.fetchFees()

                return self?.feeRepository.store(fees) ?? .empty()

    }
```

### Error Handling

```swift
class ServiceWithRetry {
    
    func fetchWithRetry<T>(
        _ operation: @escaping () -> Single<T>
    ) -> Single<T> {
        return operation()
            .retry(3)
            .catchError { error in
                // Log error
                Logger.log(error: error)
                
                // Rethrow or provide fallback
                return Single.error(error)
            }
    }
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Single Responsibility">
    Each service should have a single, well-defined purpose. Don't create god services that do everything.
  </Accordion>

  <Accordion title="Reactive APIs">
    Services should return Observable or Single for asynchronous operations, not use callbacks.
  </Accordion>

  <Accordion title="Error Handling">
    Let errors propagate through RxSwift. Don't catch and ignore errors in services.
  </Accordion>

  <Accordion title="Testability">
    Design services with interfaces (protocols) so they can be mocked for testing.
  </Accordion>

  <Accordion title="Stateless">
    Services should be stateless when possible. State belongs in repositories or models.
  </Accordion>
</AccordionGroup>
