## Overview

Screens in the Muun iOS app are implemented using view controllers that extend `MUViewController`. Each screen represents a distinct user interface with its own logic and state management.

## MUViewController

Base view controller that all screens extend:


## Screen Examples

### Home Screen

Main wallet screen showing balance and operations:

```swift
class HomeViewController: MUViewController {
    
    private let presenter: HomePresenter
    private let disposeBag = DisposeBag()
    
    // UI Components
    private var balanceView: BalanceView!
    private var operationsTableView: UITableView!
    private var sendButton: ButtonView!
    private var receiveButton: ButtonView!
    
    init(presenter: HomePresenter) {
        self.presenter = presenter
        super.init(nibName: nil, bundle: nil)
    }
    
    required init?(coder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        setUpView()
        setUpBindings()
    }
    
    private func setUpView() {
        title = "Muun"
        view.backgroundColor = Asset.Colors.background.color
        
        // Create and layout UI components
        balanceView = BalanceView()
        view.addSubview(balanceView)
        
        sendButton = ButtonView()
        sendButton.button.setTitle("Send", for: .normal)
        sendButton.button.addTarget(self, action: #selector(onSendTapped), for: .touchUpInside)
        view.addSubview(sendButton)
        
        receiveButton = ButtonView()
        receiveButton.button.setTitle("Receive", for: .normal)
        receiveButton.button.addTarget(self, action: #selector(onReceiveTapped), for: .touchUpInside)
        view.addSubview(receiveButton)
        
        operationsTableView = UITableView()
        operationsTableView.delegate = self
        operationsTableView.dataSource = self
        view.addSubview(operationsTableView)
        
        // Layout constraints...
    }
    
    private func setUpBindings() {
        // Watch user balance
        presenter.watchBalance()
            .subscribe(onNext: { [weak self] balance in
                self?.balanceView.set(balance: balance)
            })
            .disposed(by: disposeBag)
        
        // Watch operations
        presenter.watchOperations()
            .subscribe(onNext: { [weak self] operations in
                self?.operations = operations
                self?.operationsTableView.reloadData()
            })
            .disposed(by: disposeBag)
    }
    
    @objc private func onSendTapped() {
        let vc = NewOperationViewController()
        navigationController?.pushViewController(vc, animated: true)
    }
    
    @objc private func onReceiveTapped() {
        let vc = ReceiveViewController()
        navigationController?.pushViewController(vc, animated: true)
    }
}

extension HomeViewController: UITableViewDelegate, UITableViewDataSource {
    
    func tableView(_ tableView: UITableView, numberOfRowsInSection section: Int) -> Int {
        return operations.count
    }
    
    func tableView(_ tableView: UITableView, cellForRowAt indexPath: IndexPath) -> UITableViewCell {
        let cell = tableView.dequeueReusableCell(withIdentifier: "OperationCell", for: indexPath) as! OperationCell
        cell.configure(with: operations[indexPath.row])
        return cell
    }
    
    func tableView(_ tableView: UITableView, didSelectRowAt indexPath: IndexPath) {
        let operation = operations[indexPath.row]
        let vc = DetailViewController(operation: operation)
        navigationController?.pushViewController(vc, animated: true)
    }
}
```

### New Operation Screen

Send Bitcoin/Lightning payment:

```swift
class NewOperationViewController: MUViewController {
    
    private let createOperationAction: AsyncAction<Operation>
    private let disposeBag = DisposeBag()
    
    // UI Components
    private var amountInput: AmountInputView!
    private var addressInput: UITextField!
    private var feeSelector: FeeSelectorView!
    private var confirmButton: ButtonView!
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        title = "Send"
        setUpView()
        setUpBindings()
        addKeyboardObservers()
    }
    
    override func viewWillDisappear(_ animated: Bool) {
        super.viewWillDisappear(animated)
        removeKeyboardObservers()
    }
    
    private func setUpBindings() {
        // Watch action state
        createOperationAction.getState()
            .subscribe(onNext: { [weak self] state in
                self?.handleActionState(state)
            })
            .disposed(by: disposeBag)
    }
    
    private func handleActionState(_ state: ActionState<Operation>) {
        switch state.type {
        case .EMPTY:
            break
            
        case .LOADING:
            showLoading("Creating payment...")
            confirmButton.isEnabled = false
            
        case .VALUE:
            dismissLoading()
            if let operation = state.getValue() {
                showSuccess(operation)
            }
            
        case .ERROR:
            dismissLoading()
            confirmButton.isEnabled = true
            if let error = state.getError() {
                showError(error)
            }
        }
    }
    
    @objc private func onConfirmTapped() {
        guard let amount = amountInput.satoshis,
              let address = addressInput.text,
              let fee = feeSelector.selectedFee else {
            showToast(message: "Please fill all fields")
            return
        }
        
        createOperationAction.run(
            amount: amount,
            address: address,
            fee: fee
        )
    }
    
    private func showSuccess(_ operation: Operation) {
        let alert = UIAlertController(
            title: "Success",
            message: "Payment created successfully",
            preferredStyle: .alert
        )
        alert.addAction(UIAlertAction(title: "OK", style: .default) { [weak self] _ in
            self?.navigationController?.popViewController(animated: true)
        })
        present(alert, animated: true)
    }
    
    private func showError(_ error: Error) {
        let alert = UIAlertController(
            title: "Error",
            message: error.localizedDescription,
            preferredStyle: .alert
        )
        alert.addAction(UIAlertAction(title: "OK", style: .default))
        present(alert, animated: true)
    }
    
    // Keyboard handling
    @objc override func keyboardWillShow(notification: NSNotification) {
        guard let keyboardFrame = notification.userInfo?[UIResponder.keyboardFrameEndUserInfoKey] as? CGRect else {
            return
        }
        
        UIView.animate(withDuration: 0.3) {
            self.view.frame.origin.y = -keyboardFrame.height / 2
        }
    }
    
    @objc override func keyboardWillHide(notification: NSNotification) {
        UIView.animate(withDuration: 0.3) {
            self.view.frame.origin.y = 0
        }
    }
}
```

### Detail Screen

Show operation details:

```swift
class DetailViewController: MUViewController {
    
    private let operation: Operation
    private let operationRepository: OperationRepository
    private let disposeBag = DisposeBag()
    
    init(operation: Operation,
         operationRepository: OperationRepository) {
        self.operation = operation
        self.operationRepository = operationRepository
        super.init(nibName: nil, bundle: nil)
    }
    
    required init?(coder: NSCoder) {
        fatalError("init(coder:) has not been implemented")
    }
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        title = "Operation Details"
        setUpView()
        observeOperation()
    }
    
    private func setUpView() {
        // Create detail rows
        let rows = [
            createDetailRow(label: "Amount", value: formatAmount(operation.amount)),
            createDetailRow(label: "Fee", value: formatAmount(operation.fee)),
            createDetailRow(label: "Status", value: operation.status.rawValue),
            createDetailRow(label: "Date", value: formatDate(operation.creationDate))
        ]
        
        if let address = operation.receiverAddress {
            rows.append(createDetailRow(label: "Address", value: address))
        }
        
        if operation.isCancelable() {
            let cancelButton = ButtonView()
            cancelButton.button.setTitle("Cancel", for: .normal)
            cancelButton.button.addTarget(self, action: #selector(onCancelTapped), for: .touchUpInside)
            view.addSubview(cancelButton)
        }
    }
    
    private func observeOperation() {
        operationRepository.watchObject(with: operation.id!)
            .subscribe(onNext: { [weak self] updatedOperation in
                self?.updateUI(with: updatedOperation)
            })
            .disposed(by: disposeBag)
    }
    
    private func updateUI(with operation: Operation) {
        // Update status label, confirmations, etc.
    }
    
    @objc private func onCancelTapped() {
        // Show confirmation and cancel operation
    }
}
```

## Common Patterns

### Dependency Injection

```swift
class HomeViewController: MUViewController {
    
    private let presenter: HomePresenter
    private let fetchOperationsAction: AsyncAction<[Operation]>
    
    init(presenter: HomePresenter,
         fetchOperationsAction: AsyncAction<[Operation]>) {
        self.presenter = presenter
        self.fetchOperationsAction = fetchOperationsAction
        super.init(nibName: nil, bundle: nil)
    }
}

// In DependencyContainer
func homeViewController() -> HomeViewController {
    return HomeViewController(
        presenter: homePresenter(),
        fetchOperationsAction: fetchOperationsAction()
    )
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Single Responsibility">
    Each view controller should handle one screen. Don't create monolithic view controllers.
  </Accordion>

  <Accordion title="Dispose Subscriptions">
    Always use `DisposeBag` to dispose RxSwift subscriptions. This prevents memory leaks.
  </Accordion>

  <Accordion title="Weak Self">
    Use `[weak self]` in closures to prevent retain cycles.
  </Accordion>

  <Accordion title="Keyboard Management">
    Remove keyboard observers in `viewWillDisappear` to prevent crashes.
  </Accordion>

  <Accordion title="Loading States">
    Show loading indicators for async operations. Dismiss them in all cases (success, error).
  </Accordion>
</AccordionGroup>
