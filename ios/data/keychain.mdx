## Overview

The Muun iOS app uses the iOS Keychain to securely store sensitive data such as private keys, authentication tokens, and device tokens. The `KeychainRepository` provides a clean abstraction over the Keychain Services API.

## Core Operations

### Store Data

Store a string securely in the keychain:

```swift
public func store(_ string: String, at key: String) throws {
    let data = string.data(using: .utf8, allowLossyConversion: true)
    
    let query = buildQuery(for: key, forInsert: true)
    let toSet = [kSecValueData as String: data!]
    
    // Try to update first, then insert if not found
    var status = SecItemUpdate(query as CFDictionary, toSet as CFDictionary)
    if status == errSecItemNotFound {
        let addQuery = query.merging(toSet) { (k, _) in k }
        status = SecItemAdd(addQuery as CFDictionary, nil)
    }
    
    guard status == errSecSuccess else {
        throw MuunError(SecureStorage.Errors.secureStorageError)
    }
}
```

Source: `~/workspace/source/falcon/app/falcon/core/Data/KeychainRepository.swift:53`

**Features:**
- Attempts update first (idempotent)
- Falls back to insert if key doesn't exist
- Throws error on failure
- UTF-8 encoding for strings

### Retrieve Data

Retrieve a stored string:

```swift
public func get(_ key: String) throws -> String {
    var query = buildQuery(for: key, forInsert: false)
    query[kSecReturnData as String] = true
    
    var item: CFTypeRef?
    let status = SecItemCopyMatching(query as CFDictionary, &item)
    
    guard status == errSecSuccess,
        let value = item as? Data else {
        throw MuunError(SecureStorage.Errors.secureStorageError)
    }
    
    guard let ret = String(data: value, encoding: .utf8) else {
        throw MuunError(SecureStorage.Errors.invalidData)
    }
    
    return ret
}
```

### Check Existence

Check if a key exists:

```swift
func has(_ key: String) throws -> Bool {
    let query = buildQuery(for: key, forInsert: false)
    let status = SecItemCopyMatching(query as CFDictionary, nil)
    
    guard status == errSecSuccess || status == errSecItemNotFound else {
        throw MuunError(SecureStorage.Errors.secureStorageError)
    }
    
    return status == errSecSuccess
}
```

### Wipe All Data

Wipe all keychain data with optional PIN preservation:

```swift
func wipe(preservePin: Bool = false) {
    var keychainStoredKeys = getKeychainStoredKeys()
    
    if preservePin {
        do {
            keychainStoredKeys[SecureStorage.Keys.pin.rawValue] = 
                try get(SecureStorage.Keys.pin.rawValue)
        } catch {
            Logger.log(error: error)
        }
    }
    
    let dict: [NSString: Any] = [
        kSecClass: kSecClassGenericPassword,
        kSecAttrService: keyPrefix,
        kSecAttrAccessGroup: group
    ]
    
    let result = SecItemDelete(dict as CFDictionary)
    
    if result != noErr && result != errSecItemNotFound {
        Logger.log(error: NSError(domain: "wipe_secure_storage_failed", code: Int(result)))
    }
    
    // Restore preserved keys
    keychainStoredKeys.forEach {
        do {
            try self.store($0.value, at: $0.key)
        } catch {
            Logger.log(error: error)
        }
    }
}
```

## Query Building

The repository builds secure keychain queries:

```swift
private func buildQuery(for key: String, forInsert: Bool) -> [String: Any] {
    let scopedKey = tagFrom(key: key)
    var query: [String: Any] = [
        kSecClass as String: kSecClassGenericPassword,
        kSecAttrGeneric as String: scopedKey,
        kSecAttrAccount as String: scopedKey,
        kSecAttrService as String: keyPrefix,
        kSecAttrAccessGroup as String: group
    ]
    
    if forInsert {
        query[kSecAttrSynchronizable as String] = false
        query[kSecAttrAccessible as String] = kSecAttrAccessibleAlwaysThisDeviceOnly
    }
    
    return query
}

private func tagFrom(key: String) -> Data {
    return (keyPrefix + "." + key).data(using: .utf8)!
}
```

## SecureStorage Wrapper

The `SecureStorage` class provides a higher-level API:

```swift
public class SecureStorage {
    
    enum Keys: String, CaseIterable {
        case privateKey
        case encriptedUserPrivateKey
        case muunPrivateKey
        case baseKeyDerivationPath
        case pin
        case authToken
        case pinAttemptsLeft
        case passwordSalt
        case recoveryCodeSalt
        case passwordPublicKey
        case recoveryCodePublicKey
        case userKeyPublicKey
        case passwordVersionKey
        case recoveryCodeVersionKey
        case userVersionKey
    }
    
    private let keychainRepository: KeychainRepository
    
    public init(keychainRepository: KeychainRepository) {
        self.keychainRepository = keychainRepository
    }
    
    func store(_ string: String, at key: Keys) throws {
        try keychainRepository.store(string, at: key.rawValue)
    }
    
    func get(_ key: Keys) throws -> String {
        return try keychainRepository.get(key.rawValue)
    }
    
    func has(_ key: Keys) throws -> Bool {
        return try keychainRepository.has(key.rawValue)
    }
    
    func delete(_ key: Keys) {
        keychainRepository.delete(key.rawValue)
    }
    
    func wipeAll(preservePin: Bool = false) {
        keychainRepository.wipe(preservePin: preservePin)
    }
}
```

## Security Features

### Access Control

<Accordion title="Always This Device Only">
  Data is never backed up to iCloud or synced across devices. This ensures sensitive keys stay on the device.
</Accordion>

<Accordion title="App Group Sharing">
  Keychain data can be shared between the main app and extensions (e.g., notification extension) using the access group.
</Accordion>

<Accordion title="No Synchronization">
  `kSecAttrSynchronizable = false` prevents iCloud Keychain sync for sensitive data.
</Accordion>

### Best Practices

<AccordionGroup>
  <Accordion title="Store Only Strings">
    The repository stores strings only. Binary data should be hex-encoded or base64-encoded before storage.
  </Accordion>

  <Accordion title="Scoped Keys">
    All keys are prefixed with bundle ID to avoid conflicts: `io.muun.falcon.privateKey`
  </Accordion>

  <Accordion title="Error Handling">
    Always handle errors when accessing keychain. Keys may not exist or access may be denied.
  </Accordion>

  <Accordion title="Wipe on Logout">
    Call `wipeAll()` when user logs out, optionally preserving PIN for convenience.
  </Accordion>
</AccordionGroup>
