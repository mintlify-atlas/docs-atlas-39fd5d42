## Overview

The `SecureStorage` class provides a type-safe, high-level API for storing sensitive data in the iOS Keychain. It wraps `KeychainRepository` with predefined keys and consistent error handling.

## SecureStorage Class

```swift
public class SecureStorage {
    
    enum Errors: Error {
        case secureStorageError
        case invalidData
    }
    
    enum Keys: String, CaseIterable {
        case privateKey
        case encriptedUserPrivateKey
        case muunPrivateKey
        case baseKeyDerivationPath
        case pin
        case authToken
        case pinAttemptsLeft
        case passwordSalt
        case recoveryCodeSalt
        case passwordPublicKey
        case recoveryCodePublicKey
        case userKeyPublicKey
        case passwordVersionKey
        case recoveryCodeVersionKey
        case userVersionKey
    }
    
    private let keychainRepository: KeychainRepository
    
    public init(keychainRepository: KeychainRepository) {
        self.keychainRepository = keychainRepository
    }
}
```

## Usage Patterns

### Wallet Initialization

```swift
class WalletInitializer {
    let secureStorage: SecureStorage
    
    func createWallet(password: String) throws {
        // Generate keys
        let privateKey = generatePrivateKey()
        let salt = generateSalt()
        let passwordKey = deriveKey(from: password, salt: salt)
        
        // Store keys
        try secureStorage.store(privateKey.toHex(), at: .privateKey)
        try secureStorage.store(salt, at: .passwordSalt)
        try secureStorage.store(passwordKey.publicKey, at: .passwordPublicKey)
        try secureStorage.store("2", at: .passwordVersionKey)
        
        // Store derivation path
        try secureStorage.store("m/schema:1'/recovery:1'", at: .baseKeyDerivationPath)
    }
    
    func isWalletInitialized() -> Bool {
        return (try? secureStorage.has(.privateKey)) ?? false
    }
}
```

```swift PIN Verification
class PINService {
    let secureStorage: SecureStorage
    
    func verifyPIN(_ pin: String) -> Bool {
        guard let storedHash = try? secureStorage.get(.pin),
              let attemptsLeft = try? secureStorage.get(.pinAttemptsLeft).toInt() else {
            return false
        }
        
        let hash = SHA256.hash(pin)
        
        if hash == storedHash {
            // Correct PIN - reset attempts
            try? secureStorage.store("5", at: .pinAttemptsLeft)
            return true
        } else {
            // Wrong PIN - decrement attempts
            let remaining = max(0, attemptsLeft - 1)
            try? secureStorage.store("\(remaining)", at: .pinAttemptsLeft)
            
            if remaining == 0 {
                // Lock account
                secureStorage.wipeAll()
            }
            
            return false
        }
    }
}
```
</CodeGroup>

### Recovery Flow

```swift
class RecoveryService {
    let secureStorage: SecureStorage
    
    func setupRecoveryCode(_ code: String) throws {
        // Derive key from recovery code
        let salt = generateSalt()
        let recoveryKey = deriveKey(from: code, salt: salt)
        
        // Store recovery data
        try secureStorage.store(salt, at: .recoveryCodeSalt)
        try secureStorage.store(recoveryKey.publicKey, at: .recoveryCodePublicKey)
        try secureStorage.store("1", at: .recoveryCodeVersionKey)
    }
    
    func recoverWallet(recoveryCode: String) throws {
        // Get recovery salt from server or backup
        let salt = try secureStorage.get(.recoveryCodeSalt)
        
        // Derive key
        let recoveryKey = deriveKey(from: recoveryCode, salt: salt)
        
        // Verify public key matches
        let storedPublicKey = try secureStorage.get(.recoveryCodePublicKey)
        guard recoveryKey.publicKey == storedPublicKey else {
            throw RecoveryError.invalidCode
        }
        
        // Decrypt and restore private key
        let encryptedKey = try secureStorage.get(.encriptedUserPrivateKey)
        let privateKey = try decrypt(encryptedKey, with: recoveryKey)
        
        try secureStorage.store(privateKey, at: .privateKey)
    }
    
    func hasRecoveryCode() -> Bool {
        return (try? secureStorage.has(.recoveryCodePublicKey)) ?? false
    }
}
```

### Key Versioning

```swift
class KeyMigrationService {
    let secureStorage: SecureStorage
    
    func migrateKeysIfNeeded() throws {
        // Check password key version
        if let versionString = try? secureStorage.get(.passwordVersionKey),
           let version = Int(versionString) {
            
            if version < 2 {
                // Migrate from v1 to v2
                try migratePasswordKeyV1toV2()
            }
        }
        
        // Check recovery code version
        if let versionString = try? secureStorage.get(.recoveryCodeVersionKey),
           let version = Int(versionString) {
            
            if version < 2 {
                // Migrate recovery code
                try migrateRecoveryCodeV1toV2()
            }
        }
    }
    
    private func migratePasswordKeyV1toV2() throws {
        // Get old key
        let oldKey = try secureStorage.get(.passwordPublicKey)
        
        // Convert to new format
        let newKey = convertKeyFormat(oldKey)
        
        // Store new key
        try secureStorage.store(newKey, at: .passwordPublicKey)
        try secureStorage.store("2", at: .passwordVersionKey)
    }
}
```

## Integration with Database Migrations

```swift
    do {
        if try self.secureStorage.has(.passwordPublicKey) {
            try self.secureStorage.store("1", at: .passwordVersionKey)
        }
        
        if try self.secureStorage.has(.recoveryCodePublicKey) {
            try self.secureStorage.store("1", at: .recoveryCodeVersionKey)
        }
    } catch {
        Logger.log(error: error)
    }

migrator.registerMigration("move base key path to secure storage") { _ in
    if let path = self.preferences.string(forKey: .baseKeyDerivationPath) {
        do {
            try self.secureStorage.store(path, at: .baseKeyDerivationPath)
            self.preferences.remove(key: .baseKeyDerivationPath)
        } catch {
            Logger.log(error: error)
        }
    }
}
```

## Security Considerations

<AccordionGroup>
  <Accordion title="Never Log Keys">
    Never log private keys, salts, or authentication tokens. Use redacted logging for debugging.
  </Accordion>

  <Accordion title="Hex Encoding">
    Store binary data (keys) as hex strings. The keychain stores strings, so encode binary data first.
  </Accordion>

  <Accordion title="Wipe on Logout">
    Always wipe secure storage on logout. Optionally preserve PIN for convenience.
  </Accordion>

  <Accordion title="Version Keys">
    Store version numbers for cryptographic keys to enable safe migrations.
  </Accordion>

  <Accordion title="Error Recovery">
    Handle keychain errors gracefully. Data may be inaccessible due to device lock or corruption.
  </Accordion>
</AccordionGroup>

## Best Practices

1. **Type Safety**: Use the `Keys` enum instead of raw strings
2. **Error Handling**: Always handle errors when accessing secure storage
3. **Existence Checks**: Check `has()` before `get()` to avoid exceptions
4. **Atomic Operations**: Store related keys together to maintain consistency
5. **Cleanup**: Delete keys when no longer needed
6. **Testing**: Mock secure storage for unit tests
